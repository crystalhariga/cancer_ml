#set up avganncount as bins 
b <- c(-Inf, 76.0, 518.0, Inf)
names <- c("Low", "Medium", "High")
Cancer.df$avganncount.cat <- cut(Cancer.df$avganncount, breaks = b, labels = names)

# partition
set.seed(1)  
train.index <- sample(c(1:dim(Cancer.df)[1]), dim(Cancer.df)[1]*0.6)  
train.df <- Cancer.df[train.index, ]
valid.df <- Cancer.df[-train.index, ]

train.df$avganncount.cat <- as.factor(train.df$avganncount.cat)

#Randon Forest Creation 
set.seed(123)
rf <- randomForest(as.factor(avganncount.cat) ~ .-avganncount, data = train.df, ntree = 500, mtry = 4, nodesize = 5, importance = TRUE)  

#Variable Importance Plot 
varImpPlot(rf, type = 1)

#Confusion Matrix Creation 
rf.pred <- predict(rf, valid.df, type = "class")
cm <- confusionMatrix(rf.pred, as.factor(valid.df$avganncount.cat))

#Grid-Search
best_F1 <- 0
for (ntree in 500:510) {
  for (mtry in 1:9) {
    for (nodesize in 1:9) {
      rf <- randomForest(as.factor(avganncount.cat) ~ ., data = train.df, ntree = ntree, mtry = mtry, nodesize = nodesize, importance = TRUE)
      rf.pred <- predict(rf, valid.df, type = "class")
      cm <- confusionMatrix(rf.pred, as.factor(valid.df$avganncount.cat))
      P <- cm$byClass[1]
      R <- cm$byClass[2]
      current_F1 <- (2*P*R) / (P+R)
      if (current_F1 > best_F1) {
        best_F1 <- current_F1
        bestParm1 <- ntree
        bestParm2 <- mtry
        bestParm3 <- nodesize
      }}}}


#Random Forest using best parameters
rf <- randomForest(as.factor(avganncount.cat) ~ .-avganncount, data = train.df, ntree = bestParm1, mtry = bestParm2, nodesize = bestParm3, importance = TRUE)
varImpPlot(rf, type = 1)

rf.pred <- predict(rf, valid.df, type = "class")
cm1 <- confusionMatrix(rf.pred, as.factor(valid.df$avganncount.cat))
