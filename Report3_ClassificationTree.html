Cancer.df <- cancer_hcluster
Cancer.df <- cancer_hcluster[-c(1)]

#determine Q1 and Q3 to make cutoffs for bins 
summary(Cancer.df$avganncount)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##     6.0    76.0   171.0   606.3   518.0 38150.0
#set up avganncount as bins 
b <- c(-Inf, 76.0, 518.0, Inf)
names <- c("Low", "Medium", "High")
Cancer.df$avganncount.cat <- cut(Cancer.df$avganncount, breaks = b, labels = names)

# predict & Confusion matrix
set.seed(1)
train.index <- sample(c(1:dim(Cancer.df)[1]), dim(Cancer.df)[1]*0.6)  
train.df <- Cancer.df[train.index, ]
valid.df <- Cancer.df[-train.index, ]

#monitor depth 
default.ct <- rpart(avganncount.cat ~ .-avganncount, data = train.df, method = "class", 
                    control = rpart.control(maxdepth = 20))
prp(default.ct, type = 1, extra = 1, under = TRUE, split.font = 1, varlen = -10)

#monitor cost of complexity
deeper.ct <- rpart(avganncount.cat ~ .-avganncount, data = train.df, method = "class",
                   cp = 0, minsplit = 1)

#count number of leaves
length(default.ct$frame$var[default.ct$frame$var == "<leaf>"])
## [1] 3
length(deeper.ct$frame$var[deeper.ct$frame$var == "<leaf>"])
## [1] 185
#Testing on data after partitioning 
default.ct.point.pred.train <- predict(default.ct,train.df,type = "class")

# matrix 1 - train
default.ct.point.pred.train <- predict( default.ct, train.df,type = "class")
cm1 <- confusionMatrix( default.ct.point.pred.train, as.factor(train.df$avganncount.cat))

accuracy <- cm1$overall[1]
Sensitivity <- cm1$byClass[1] 
Specificity <- cm1$byClass[2] 
F1 <- (2 * Sensitivity * Specificity) / (Sensitivity + Specificity)
cat("Classification Tree accuracy=", accuracy, " and ", " F1=", F1)
## Classification Tree accuracy= 0.8977024  and   F1= 0.9519547
# matrix 2 - validation
default.ct.point.pred.valid <- predict( default.ct, valid.df, type = "class")
cm1_valid <- confusionMatrix( default.ct.point.pred.valid, as.factor(valid.df$avganncount.cat))

accuracy <- cm1_valid$overall[1]
Sensitivity <- cm1_valid$byClass[1] 
Specificity <- cm1_valid$byClass[2] 
F1 <- (2 * Sensitivity * Specificity) / (Sensitivity + Specificity)
cat("Classification Tree accuracy=", accuracy, " and ", " F1=", F1)
## Classification Tree accuracy= 0.8835111  and   F1= 0.943235
# matrix 3 - deeper tree - train the tree model
deeper.ct.point.pred.train <- predict( deeper.ct, train.df, type = "class")
cm2 <- confusionMatrix( deeper.ct.point.pred.train, as.factor(train.df$avganncount.cat) )

accuracy <- cm2$overall[1]
Sensitivity <- cm2$byClass[1] # where did this come from?
Specificity <- cm2$byClass[2] 
F1 <- (2 * Sensitivity * Specificity) / (Sensitivity + Specificity)
cat("Classification Tree accuracy=", accuracy, " and ", " F1=", F1)
## Classification Tree accuracy= 1  and   F1= 1
# matrix 4 - deeper tree - validation to compare predicted value against actual value in the validation data
deeper.ct.point.pred.valid <- predict( deeper.ct, valid.df, type = "class")
cm2_valid <- confusionMatrix( deeper.ct.point.pred.valid, as.factor(valid.df$avganncount.cat) )

accuracy <- cm2_valid$overall[1]
Sensitivity <- cm2_valid$byClass[1] # where did this come from?
Specificity <- cm2_valid$byClass[2] 
F1 <- (2 * Sensitivity * Specificity) / (Sensitivity + Specificity)
cat("Classification Tree accuracy=", accuracy, " and ", " F1=", F1)
## Classification Tree accuracy= 0.8490566  and   F1= 0.8656969
#Gridsearch for best  hyperparameters 
curr_F1 <- 0  

for( cost_penalty in seq(from=0, to=0.1, by=0.01)) {
  for( min_leaf_to_split in seq(from=1, to=10, by=1)) {
    
    # train the tree
    trained_tree <- rpart(avganncount.cat ~ .-avganncount, data = train.df, method = "class", 
                          cp = cost_penalty, minsplit = min_leaf_to_split)
    
    # predict with the trained tree
    train.results <- predict( trained_tree, train.df, type = "class" )
    valid.results <- predict( trained_tree, valid.df, type = "class" )  
    
    # generate the confusion matrix to compare the prediction with the actual value of Personal Loan acceptance (0/1), 
    # to calculate the sensitivity and specificity
    results <- confusionMatrix( valid.results, as.factor(valid.df$avganncount.cat) )
    
    # calculate F1 from results
    Sensitivity <- results$byClass[1] # where did this come from?
    Specificity <- results$byClass[2] 
    F1 <- (2 * Sensitivity * Specificity) / (Sensitivity + Specificity)
    
    # Is this F1 the best we have so far? If so, store the current values:
    if( F1 > curr_F1 ) {
      curr_F1 <- F1
      best_cost_penalty <- cost_penalty
      best_min_leaf_to_split <- min_leaf_to_split
    }
  }
}

#best parameters according to gridsearch
cat("best F1=" , curr_F1, "; best best_cost_penalty", best_cost_penalty, "; best_min_leaf_to_split=", best_min_leaf_to_split)
## best F1= 0.943235 ; best best_cost_penalty 0.01 ; best_min_leaf_to_split= 1
# retrain the tree to match the best parameters we found  
trained_tree <- rpart(avganncount.cat ~ .-avganncount, data = train.df, method = "class", 
                      cp = best_cost_penalty , minsplit = best_min_leaf_to_split )  
prp(trained_tree, type = 1, extra = 1, under = TRUE, split.font = 1, varlen = -10, 
    box.col=ifelse(trained_tree$frame$var == "<leaf>", 'gray', 'white'))
